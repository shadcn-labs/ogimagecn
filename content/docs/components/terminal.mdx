## Usage

```tsx
import { ImageResponse } from "next/og";

import { Terminal } from "@/components/og/terminal";

export function GET() {
  return new ImageResponse(
    <Terminal
      brand="acme"
      title="Now in public beta"
      caption="npm i @acme/sdk"
    />,
    { width: 1200, height: 630 }
  );
}
```
