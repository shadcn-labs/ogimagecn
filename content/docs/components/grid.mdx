## Usage

```tsx
import { ImageResponse } from "next/og";

import { Grid } from "@/components/og/grid";

export function GET() {
  return new ImageResponse(<Grid title="Build your own OG images" />, {
    width: 1200,
    height: 630,
  });
}
```
