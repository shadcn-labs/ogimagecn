## Usage

```tsx
import { ImageResponse } from "next/og";

import { Shiori } from "@/components/og/shiori";

export function GET() {
  return new ImageResponse(
    <Shiori
      title="Shiori"
      description="A beautifully simple read-it-later app"
      accent="#f97316"
      accentSecondary="#ea580c"
    />,
    { width: 1200, height: 630 },
  );
}
```
