## Usage

```tsx
import { ImageResponse } from "next/og";

import { Showcase } from "@/components/og/showcase";

export function GET() {
  return new ImageResponse(
    <Showcase
      title="Run your business smarter"
      subtitle="Every metric in one calm view."
      url="app.acme.com"
    />,
    { width: 1200, height: 630 }
  );
}
```
