## Usage

```tsx
import { ImageResponse } from "next/og";

import { Event } from "@/components/og/event";

export function GET() {
  return new ImageResponse(
    <Event
      label="Webinar"
      title="Shipping beautiful OG images at scale"
      date="Jun 5, 2026 · 10:00 AM PT"
      location="Online"
    />,
    { width: 1200, height: 630 }
  );
}
```
