## Usage

```tsx
import { ImageResponse } from "next/og";

import { Editorial } from "@/components/og/editorial";

export function GET() {
  return new ImageResponse(
    <Editorial
      kicker="Feature"
      title="Designing at the edge of the canvas"
      author="The OG Times"
      meta="Issue 04"
    />,
    { width: 1200, height: 630 }
  );
}
```
