## Usage

```tsx
import { ImageResponse } from "next/og";

import { Profile } from "@/components/og/profile";

export function GET() {
  return new ImageResponse(
    <Profile
      name="Ada Lovelace"
      role="Founder & Engineer"
      website="ada.dev"
      avatar="https://example.com/avatar.png"
    />,
    { width: 1200, height: 630 }
  );
}
```
