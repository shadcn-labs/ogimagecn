## Usage

```tsx
import { ImageResponse } from "next/og";

import { ShadcnRegistry11 } from "@/components/og/shadcn-registry-11";

export function GET() {
  return new ImageResponse(
    <ShadcnRegistry11
      title="Landing page components built with React, Shadcn/ui and Tailwind that you can copy/paste into your project."
      brand="Launch UI"
      techIcons={["React", "Shadcn", "Tailwind", "Figma", "TypeScript"]}
    />,
    {
      width: 1200,
      height: 630,
    }
  );
}
```
